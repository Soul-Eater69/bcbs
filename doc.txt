import json
from pathlib import Path

src = Path("output/stage_prediction_eval/stage_prediction_dataset_gt_patched.json")
out = Path("output/stage_prediction_eval/tickets_with_stage_gt.txt")

data = json.loads(src.read_text(encoding="utf-8"))
ids = []
for ticket_id, row in (data.get("tickets") or {}).items():
    gt = ((row.get("ground_truth") or {}).get("gt_by_value_stream") or {})
    if gt:
        ids.append(ticket_id)

out.write_text("\n".join(ids) + "\n", encoding="utf-8")
print(f"wrote {len(ids)} tickets to {out}")
